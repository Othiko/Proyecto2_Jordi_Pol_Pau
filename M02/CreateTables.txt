create database batallaRaces;

use batallaRaces;

CREATE TABLE Personatges (
	id_personatge int not null auto_increment,
	nom_personatge char(30),
	raca char(10),
	imagePath_personatge char(80),
	vida int,
	forca int,
	defensa int,
	agilitat int,
	velocitat int,
	puntuacioGlobal int,
	enemicsDerrotats int,
	danyRealitzat int,
	danyRebut int,
	PRIMARY KEY (id_personatge)
);

CREATE TABLE Armes (
	id_arma int not null auto_increment,
	nom_arma char(30),
	velocitat_arma int,
	forca_arma int,
	huma boolean,
	elf boolean,
	nan boolean,
	PRIMARY KEY (id_arma)
);
